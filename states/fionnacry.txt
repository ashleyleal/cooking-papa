from states.state import State
from button import Button
from assets.assets import * 

class Shop_Menu(State):

    def __init__(self, game):
        State.__init__(self, game)

    def update(self, actions):
        super().update(actions)
        if actions["menu"]:
            main_menu = self.game.state_stack[0]
            main_menu.enter_state()

def render(self, surface):
  Characters_button = Button(x position, y position, image name, 1)
  Wallpaper_button = Button(x position, y position, image name, 1)
  Music_button = Button(x position, y position, image name, 1)

  self.game.draw_image(Shop_bg, 1, surface, self.game.GAME_X / 2, self.game.GAME_Y / 2 )
  self.game.draw_image(Characters_button, 1, surface, self.game.GAME_X (plus, minus, or divide),self.game.GAME_Y(plus, minus, or divide))
  self.game.draw_image(Wallpapper_button, 1, surface, self.game.GAME_X (plus, minus, or divide), self.game.GAME_Y (plus, minus, or divide))
  self.game.draw_image(Music_button, 1, surface, self.game.GAME_X (plus, minus, or divide), self.game.GAME_Y (plus, minus, or divide))

  if self.character_button.draw(surface):
    self.game.draw_image(Shop_bg, 1, surface, self.game.GAME_X / 2, self.game.GAME_Y / 2 )
